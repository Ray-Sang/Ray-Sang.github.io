Lecture 14, Lab 5, case-select statement

2020-10-08
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity case_sel is
port(
	A : in std_logic;
	B : in std_logic;
	C : in std_logic;
	D : in std_logic;
	sel : in std_logic_vector(1 downto 0);
	Y : out std_logic
);
end case_sel;


architecture Behavioral of case_sel is

begin
process(A, B, C, D, sel)
	begin
		case sel is
			when "00" => Y <= A;
			when "01" => Y <= B;
			when "10" => Y <= C;
			when "11" => Y <= D;
			when others => Y <= '0';
		end case;
	end process;
end Behavioral;
